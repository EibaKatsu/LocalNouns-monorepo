# nouns-monorepoからclone
git clone https://github.com/nounsDAO/nouns-monorepo.git

# リモートリポジトリ参照
git remote -v

# リモートリポジトリを削除
git remote remove origin

# githubで自分のリモートリポジトリを作成
https://github.com/EibaKatsu/LocalNouns-monorepo

# リモートリポジトリの切り替え
git remote add origin https://github.com/EibaKatsu/LocalNouns-monorepo.git

# ブランチの作成
git branch -M main

# ブランチにpush
git push -u origin main

# ブランチを作成して切り替え
git checkout -b eiba20230507_01

# リモートリポジトリへ共有
git push -u origin eiba20230507_01

# .envの作成
略

# hardhatローカル起動
npx hardhat node

# コンパイル
yarn build:sol

# SVG生成コントラクトデプロイのtask
npx hardhat deploy-descriptor-v2 --dao-executor 0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266 --network localhost
npx hardhat deploy-descriptor-v2 --dao-executor 0x52A76a606AC925f7113B4CC8605Fe6bCad431EbB --network goerli

# populate (テスト用)
npx hardhat populate-descriptor --nft-descriptor 0x97c82f1c82091F9fC86bd2008b48286d14A38D15 --nouns-descriptor 0x9E6aC85469E8E6ee53ACE807C967f60ACa6B9555 --network goerli
